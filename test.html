body { padding: 0; margin: 0; } html, body, #map { height: 100%; width: 100vw; } var map = L.map('map').fitWorld(); L.tileLayer('https://api.mapbox.com/styles/v1/{id}/tiles/{z}/{x}/{y}?access_token={accessToken}', { attribution: 'Map data © OpenStreetMap contributors, Imagery Â© Mapbox', maxZoom: 18, tileSize: 512, zoomOffset: -1 }).addTo(map); map.locate({setView: true, maxZoom: 16}); function onLocationFound(e) { var radius = e.accuracy; L.marker(e.latlng).addTo(map) .bindPopup("You are within " + radius + " meters from this point").openPopup(); L.circle(e.latlng, radius).addTo(map); } map.on('locationfound', onLocationFound); function onLocationError(e) { alert(e.message); } map.on('locationerror', onLocationError);
